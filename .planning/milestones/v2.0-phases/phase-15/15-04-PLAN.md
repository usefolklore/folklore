/**
 * Phase 15: Peer Foundation + Security — requirement tests.
 *
 * Covers PEER-01..05, SEC-01..06.
 * Pure unit tests except PEER-01 (uses tmp dir for identity file).
 */
import assert from 'node:assert/strict';
import { test, describe } from 'node:test';
import { mkdtempSync, rmSync, existsSync, readFileSync } from 'node:fs';
import { tmpdir } from 'node:os';
import { join } from 'node:path';

import { scanNode, auditRoom, buildPatterns } from '../src/domain/sharing.js';
import type { ShareableNode } from '../src/domain/sharing.js';
import type { GraphNode } from '../src/domain/graph.js';
import { loadOrCreateIdentity } from '../src/infrastructure/peer-transport.js';
import { loadPeers, savePeers, addPeerRecord, removePeerRecord } from '../src/infrastructure/peer-store.js';
import type { PeersFile, PeerRecord } from '../src/infrastructure/peer-store.js';

// ─────────── helpers ───────────

const makeNode = (overrides: Partial<GraphNode> = {}): GraphNode => ({
  id: overrides.id ?? 'test-node-1',
  label: overrides.label ?? 'Clean test node',
  file_type: 'document',
  source_file: '/tmp/test.md',
  room: overrides.room ?? 'test-room',
  source_uri: overrides.source_uri,
  fetched_at: overrides.fetched_at,
  embedding_id: overrides.embedding_id,
  ...overrides,
});

const patterns = buildPatterns();

// ─────────── SEC-01: secrets scanner detects all 10 patterns ───────────

describe('SEC-01: secrets scanner', () => {
  const secretSamples: Array<{ patternName: string; field: 'label' | 'source_uri'; value: string }> = [
    // Test values built via runtime concatenation so GitHub secret scanner
    // doesn't flag this plan doc. Each still matches its corresponding regex.
    { patternName: 'openai-key',    field: 'label',      value: 'key is sk-' + 'abcdefghij1234567890xx' },
    { patternName: 'github-token',  field: 'label',      value: 'token ghp_' + 'aBcDeFgHiJkLmNoPqRsTuVwXyZ012345678a' },
    { patternName: 'github-oauth',  field: 'label',      value: 'oauth gho_' + 'aBcDeFgHiJkLmNoPqRsTuVwXyZ012345678a' },
    { patternName: 'aws-key-id',    field: 'source_uri', value: 'https://s3.aws?key=AKIA' + 'IOSFODNN7EXAMPLE' },
    { patternName: 'stripe-live',   field: 'label',      value: 'sk_live_' + 'abcdefghijklmnopqrstuvwx' },
    { patternName: 'bearer-token',  field: 'label',      value: 'Bearer ' + 'eyJhbGciOiJIUzI1NiIsI' },
    { patternName: 'password-kv',   field: 'label',      value: 'password=' + 'mysecretpassword123' },
    { patternName: 'api-key-kv',    field: 'label',      value: 'api_key=' + 'abcdefghij1234567890' },
    { patternName: 'env-token',     field: 'label',      value: 'GITHUB_TOKEN=ghp_' + 'realvalue123456' },
    { patternName: 'env-secret',    field: 'label',      value: 'AWS_SECRET=AKIA' + 'IOSFODNN7EXAMPL1' },
  ];

  for (const { patternName, field, value } of secretSamples) {
    test(`detects ${patternName} in ${field}`, () => {
      const node = makeNode({ [field]: value });
      const result = scanNode(node, patterns);
      assert.ok(result.isErr(), `expected err for ${patternName}`);
      const match = result.error.matches.find((m) => m.patternName === patternName);
      assert.ok(match, `expected match for pattern ${patternName}, got: ${JSON.stringify(result.error.matches)}`);
      assert.equal(match.field, field);
    });
  }

  test('buildPatterns returns 10 built-in patterns', () => {
    assert.equal(patterns.length, 10);
  });

  test('buildPatterns merges custom patterns', () => {
    const custom = buildPatterns([{ name: 'custom', pattern: 'CUSTOM_\\d+' }]);
    assert.equal(custom.length, 11);
    assert.equal(custom[10].name, 'custom');
  });
});

// ─────────── SEC-02: flagged nodes are BLOCKED ───────────

describe('SEC-02: flagged nodes blocked', () => {
  test('scanNode returns err with SecretDetected type', () => {
    const node = makeNode({ label: 'has sk-abcdefghij1234567890xx' });
    const result = scanNode(node, patterns);
    assert.ok(result.isErr());
    assert.equal(result.error.type, 'SecretDetected');
    assert.equal(result.error.nodeId, node.id);
    assert.ok(result.error.matches.length > 0);
  });

  test('auditRoom counts blocked nodes correctly', () => {
    const clean1 = makeNode({ id: 'c1' });
    const clean2 = makeNode({ id: 'c2' });
    const flagged = makeNode({ id: 'f1', label: 'has sk-abcdefghij1234567890xx' });
    const result = auditRoom([clean1, clean2, flagged], patterns);
    assert.equal(result.allowed.length, 2);
    assert.equal(result.blocked.length, 1);
    assert.equal(result.blocked[0].nodeId, 'f1');
  });
});

// ─────────── SEC-03: shared nodes carry only safe fields ───────────

describe('SEC-03: ShareableNode field boundary', () => {
  test('scanNode ok result has no file_type or source_file', () => {
    const node = makeNode({ id: 'safe-1', label: 'harmless' });
    const result = scanNode(node, patterns);
    assert.ok(result.isOk());
    const keys = Object.keys(result.value);
    assert.ok(!keys.includes('file_type'), 'ShareableNode must not have file_type');
    assert.ok(!keys.includes('source_file'), 'ShareableNode must not have source_file');
  });

  test('scanNode ok result has expected fields', () => {
    const node = makeNode({
      id: 'safe-2',
      label: 'test',
      room: 'r1',
      source_uri: 'https://example.com',
      fetched_at: '2026-01-01T00:00:00Z',
      embedding_id: 'emb-1',
    });
    const result = scanNode(node, patterns);
    assert.ok(result.isOk());
    assert.equal(result.value.id, 'safe-2');
    assert.equal(result.value.label, 'test');
    assert.equal(result.value.room, 'r1');
    assert.equal(result.value.source_uri, 'https://example.com');
    assert.equal(result.value.fetched_at, '2026-01-01T00:00:00Z');
    assert.equal(result.value.embedding_id, 'emb-1');
  });
});

// ─────────── SEC-04: share audit shows what would be shared ───────────

describe('SEC-04: auditRoom', () => {
  test('returns allowed and blocked arrays', () => {
    const nodes = [
      makeNode({ id: 'a1', label: 'safe' }),
      makeNode({ id: 'a2', label: 'also safe' }),
      makeNode({ id: 'b1', label: 'ghp_' + 'aBcDeFgHiJkLmNoPqRsTuVwXyZ012345678a' }),
    ];
    const result = auditRoom(nodes, patterns);
    assert.equal(result.allowed.length, 2);
    assert.equal(result.blocked.length, 1);
    assert.equal(result.blocked[0].nodeId, 'b1');
    assert.ok(result.blocked[0].matches.some((m) => m.patternName === 'github-token'));
  });

  test('all-clean room returns all allowed, zero blocked', () => {
    const nodes = [makeNode({ id: 'c1' }), makeNode({ id: 'c2' })];
    const result = auditRoom(nodes, patterns);
    assert.equal(result.allowed.length, 2);
    assert.equal(result.blocked.length, 0);
  });

  test('empty room returns empty arrays', () => {
    const result = auditRoom([], patterns);
    assert.equal(result.allowed.length, 0);
    assert.equal(result.blocked.length, 0);
  });
});

// ─────────── SEC-05: noise encryption (structural — contiguous pattern) ───────────

describe('SEC-05: Noise encryption', () => {
  test('peer-transport.ts configures connectionEncrypters: [noise()] as contiguous pattern', () => {
    const src = readFileSync(join(import.meta.dirname, '..', 'src', 'infrastructure', 'peer-transport.ts'), 'utf8');
    // Assert the contiguous pattern — not just separate occurrences of noise() and connectionEncrypters
    assert.match(src, /connectionEncrypters:\s*\[noise\(\)\]/, 'peer-transport.ts must configure connectionEncrypters: [noise()] as a contiguous pattern');
  });
});

// ─────────── SEC-06: peer auth via ed25519 (12D3 prefix) ───────────

describe('SEC-06: peer authentication', () => {
  test('PeerId derived from ed25519 identity starts with 12D3 prefix', async () => {
    const tmp = mkdtempSync(join(tmpdir(), 'wi-sec06-'));
    try {
      const idPath = join(tmp, 'peer-identity.json');
      const result = await loadOrCreateIdentity(idPath);
      assert.ok(result.isOk(), `expected ok, got: ${JSON.stringify(result)}`);
      assert.ok(result.value.peerId.length > 10, 'PeerId should be a substantial string');
      assert.ok(typeof result.value.peerId === 'string');
      // ed25519 PeerIds in libp2p use base32-encoded CIDv1 with identity multihash,
      // which produces a string starting with '12D3' (base58btc) or similar prefix
      assert.ok(
        result.value.peerId.startsWith('12D3'),
        `PeerId should start with 12D3 prefix (ed25519 multihash), got: ${result.value.peerId.slice(0, 10)}...`,
      );
    } finally {
      rmSync(tmp, { recursive: true, force: true });
    }
  });
});

// ─────────── PEER-01: ed25519 keypair generation ───────────

describe('PEER-01: identity generation', () => {
  test('generates peer-identity.json when absent', async () => {
    const tmp = mkdtempSync(join(tmpdir(), 'wi-peer01-'));
    try {
      const idPath = join(tmp, 'peer-identity.json');
      assert.ok(!existsSync(idPath));
      const result = await loadOrCreateIdentity(idPath);
      assert.ok(result.isOk());
      assert.ok(existsSync(idPath), 'peer-identity.json must be created');

      const stored = JSON.parse(readFileSync(idPath, 'utf8'));
      assert.ok(stored.privateKeyB64, 'must have privateKeyB64');
      assert.ok(stored.peerId, 'must have peerId');
      assert.ok(stored.createdAt, 'must have createdAt');
    } finally {
      rmSync(tmp, { recursive: true, force: true });
    }
  });

  test('loads existing identity with stable PeerId', async () => {
    const tmp = mkdtempSync(join(tmpdir(), 'wi-peer01b-'));
    try {
      const idPath = join(tmp, 'peer-identity.json');
      const r1 = await loadOrCreateIdentity(idPath);
      assert.ok(r1.isOk());
      const r2 = await loadOrCreateIdentity(idPath);
      assert.ok(r2.isOk());
      assert.equal(r1.value.peerId, r2.value.peerId, 'PeerId must be stable across loads');
    } finally {
      rmSync(tmp, { recursive: true, force: true });
    }
  });

  test('private key raw is 64 bytes (ed25519)', async () => {
    const tmp = mkdtempSync(join(tmpdir(), 'wi-peer01c-'));
    try {
      const idPath = join(tmp, 'peer-identity.json');
      const result = await loadOrCreateIdentity(idPath);
      assert.ok(result.isOk());
      assert.equal(result.value.privateKey.raw.byteLength, 64, 'ed25519 raw key = 32 priv + 32 pub = 64 bytes');
    } finally {
      rmSync(tmp, { recursive: true, force: true });
    }
  });
});

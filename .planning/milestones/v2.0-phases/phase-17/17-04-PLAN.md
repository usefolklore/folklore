       const buildFakeVectorIndex = (records: readonly VectorRecord[]): VectorIndex => {
         const byId = new Map(records.map((r) => [r.node_id, r]));
         const l2 = (a: Vector, b: Vector): number => {
           let s = 0; for (let i = 0; i < a.length; i++) { const d = a[i]-b[i]; s += d*d; } return Math.sqrt(s);
         };
         return {
           upsert: (r) => { byId.set(r.node_id, r); return okAsync(undefined); },
           searchGlobal: (q, k) =>
